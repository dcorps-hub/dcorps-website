#!/usr/bin/env bash
#
# extract_debug.sh
#
# Recursively extracts content from all files & folders (except node_modules),
# skipping package-lock.json / yarn.lock, and skipping the output file itself.
# Includes detailed debug output with `set -x`.

# Turn on shell debugging (prints each command as it executes)
set -x

OUTFILE="extracted_content.txt"

# Clear/create the output file
echo "Extraction of files (excluding node_modules)..." > "$OUTFILE"
echo "-----------------------------------------------" >> "$OUTFILE"

function process_dir() {
  local dir="$1"
  echo "Processing directory: $dir"

  for entry in "$dir"/*; do
    [ -e "$entry" ] || continue
    echo "  Found: $entry"

    # Skip node_modules folder
    if [[ "$entry" == *"/node_modules"* ]]; then
      echo "  --> Skipping node_modules: $entry"
      continue
    fi

    # Skip the output file itself to prevent infinite re-append
    if [[ "$entry" == "./$OUTFILE" ]]; then
      echo "  --> Skipping output file: $entry"
      continue
    fi

    # Skip lock files that mention node_modules in plain text
    if [[ "$entry" == *"package-lock.json" ]] || [[ "$entry" == *"yarn.lock" ]]; then
      echo "  --> Skipping lock file: $entry"
      continue
    fi

    # If it's a directory, recurse
    if [ -d "$entry" ]; then
      process_dir "$entry"
    else
      # It's a file; decide if binary or text
      case "$entry" in
        *.png|*.jpg|*.jpeg|*.gif|*.pdf|*.svg|*.ico|*.mp3|*.mp4|*.mov|*.avi|*.zip|*.tar|*.gz|*.exe)
          echo "  --> BINARY file, listing only: $entry"
          echo "----- File: $entry (BINARY FILE) -----" >> "$OUTFILE"
          ;;
        *)
          echo "  --> TEXT file, dumping contents: $entry"
          echo "----- File: $entry -----" >> "$OUTFILE"
          cat "$entry" >> "$OUTFILE"
          echo "" >> "$OUTFILE"
          ;;
      esac
    fi
  done
}

# Start from current directory
process_dir "."

echo "-----------------------------------------------" >> "$OUTFILE"
echo "Extraction complete! Check $OUTFILE." >> "$OUTFILE"
echo "Done."

  // Add this local state for toggling the Roadmap’s open/collapsed
  const [isOpen, setIsOpen] = useState(true);

  useEffect(() => {

function truncateTask(task, limit = 20) {
  if (task.length > limit) {
    return task.substring(0, limit) + '...';
  }
  return task;
}

function RoadmapList({ isOpen, setIsOpen }) {
